    for (long i = 0; i < N; i++) {
        arr[i] = rand() % 100;
    }

    long count = 0;

    for (long i = 0; i < N; i++) {

    return 0;
}




				count_pthread.c




#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <time.h>

#define N 100000000
#define TARGET 42
#define NUM_THREADS 4

int *arr;
long global_count = 0;
pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;

typedef struct {
    long start;
    long end;
} ThreadArgs;

void* count_chunk(void* arg) {
    ThreadArgs *args = (ThreadArgs*) arg;
    long local_count = 0;

    for (long i = args->start; i < args->end; i++)
        if (arr[i] == TARGET)
            local_count++;

    pthread_mutex_lock(&mutex);
    global_count += local_count;
    pthread_mutex_unlock(&mutex);

    return NULL;
}

int main() {
    arr = malloc(N * sizeof(int));
    if (!arr) { perror("malloc"); return 1; }

    srand(time(NULL));
    for (long i = 0; i < N; i++)
        arr[i] = rand() % 100;

    pthread_t threads[NUM_THREADS];
    ThreadArgs args[NUM_THREADS];
    long chunk_size = N / NUM_THREADS;

    clock_t start = clock();

    for (int t = 0; t < NUM_THREADS; t++) {
        args[t].start = t * chunk_size;
        args[t].end = (t == NUM_THREADS - 1) ? N : (t + 1) * chunk_size;
        pthread_create(&threads[t], NULL, count_chunk, &args[t]);
    }

    for (int t = 0; t < NUM_THREADS; t++)
        pthread_join(threads[t], NULL);

    clock_t end = clock();
    double elapsed = (double)(end - start) / CLOCKS_PER_SEC;

    printf("Count = %ld\n", global_count);
    printf("Time with %d threads = %.2f seconds\n", NUM_THREADS, elapsed);

    free(arr);
    pthread_mutex_destroy(&mutex);
    return 0;
}

					count_threads.py




import random
import time
import threading

N = 100_000_000
TARGET = 42
NUM_THREADS = 4

arr = [random.randint(0, 99) for _ in range(N)]
global_count = 0
lock = threading.Lock()

def count_chunk(start, end):
    global global_count
    local_count = 0
    for i in range(start, end):
        if arr[i] == TARGET:
            local_count += 1
    with lock:
        global_count += local_count

threads = []
chunk_size = N // NUM_THREADS

start_time = time.time()

for t in range(NUM_THREADS):
    s = t * chunk_size
    e = N if t == NUM_THREADS - 1 else (t + 1) * chunk_size
    thread = threading.Thread(target=count_chunk, args=(s, e))
    threads.append(thread)
    thread.start()

for t in threads:
    t.join()

end_time = time.time()

print(f"Count = {global_count}")
print(f"Time = {end_time - start_time:.2f} seconds")

import random
import time
import multiprocessing as mp

N = 100_000_000
TARGET = 42
NUM_PROCESSES = 4

arr = [random.randint(0, 99) for _ in range(N)]

def count_chunk(start, end):
    local_count = 0
    for i in range(start, end):
        if arr[i] == TARGET:
            local_count += 1
    return local_count

if __name__ == "__main__":
    chunk_size = N // NUM_PROCESSES
    pool = mp.Pool(NUM_PROCESSES)
    results = []

    start_time = time.time()

    for t in range(NUM_PROCESSES):
        s = t * chunk_size
        e = N if t == NUM_PROCESSES - 1 else (t + 1) * chunk_size
        results.append(pool.apply_async(count_chunk, (s, e)))

    pool.close()
    pool.join()

    total = sum(r.get() for r in results)

    end_time = time.time()

    print(f"Count = {total}")
    print(f"Time = {end_time - start_time:.2f} seconds")

#include <mpi.h>
#include <stdio.h>
#include <stdlib.h>

#define PING_PONG_LIMIT 10000

int main(int argc, char** argv) {

    MPI_Init(&argc, &argv);

    int world_rank;
    MPI_Comm_rank(MPI_COMM_WORLD, &world_rank);

    int world_size;
    MPI_Comm_size(MPI_COMM_WORLD, &world_size);

    if (world_size != 2) {
        if (world_rank == 0)
            printf("This program requires exactly 2 processes.\n");
        MPI_Abort(MPI_COMM_WORLD, 1);
    }

    int ping_pong_count = 0;
    int partner_rank = (world_rank == 0) ? 1 : 0;

    double start_time = MPI_Wtime();

    while (ping_pong_count < PING_PONG_LIMIT) {
        if (world_rank == ping_pong_count % 2) {
            ping_pong_count++;
            MPI_Send(&ping_pong_count, 1, MPI_INT, partner_rank, 0, MPI_COMM_WORLD);
        } else {
            MPI_Recv(&ping_pong_count, 1, MPI_INT, partner_rank, 0, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
        }
    }

    double end_time = MPI_Wtime();

    if (world_rank == 0) {
        printf("Ping-pong completed in %f seconds\n", end_time - start_time);
        printf("Average round-trip time: %f microseconds\n",
               (end_time - start_time) * 1e6 / PING_PONG_LIMIT);
    }

    MPI_Finalize();
    return 0;
}

#include <mpi.h>
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define N 100000000
#define TARGET 42

int main(int argc, char** argv) {

    int rank, size;
    long local_count = 0;
    long global_count = 0;
    double start, end;

    MPI_Init(&argc, &argv);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    long elements_per_proc = N / size;
    long remainder = N % size;
    long local_n;

    if (rank < remainder)
        local_n = elements_per_proc + 1;
    else
        local_n = elements_per_proc;

    int *local_arr = (int*) malloc(local_n * sizeof(int));
    if (!local_arr) {
        printf("Malloc failed\n");
        MPI_Abort(MPI_COMM_WORLD, 1);
    }

    srand(time(NULL) + rank);

    for (long i = 0; i < local_n; i++)
        local_arr[i] = rand() % 100;

    MPI_Barrier(MPI_COMM_WORLD);
    start = MPI_Wtime();

    for (long i = 0; i < local_n; i++)
        if (local_arr[i] == TARGET)
            local_count++;

    MPI_Reduce(&local_count, &global_count, 1, MPI_LONG, MPI_SUM, 0, MPI_COMM_WORLD);

    end = MPI_Wtime();

    if (rank == 0) {
        printf("Total count = %ld\n", global_count);
        printf("Time with %d processes = %f seconds\n", size, end - start);
    }

    free(local_arr);
    MPI_Finalize();
    return 0;
}

# Lab 1: MPI – Parallel Counting

## Overview
This lab introduces **MPI (Message Passing Interface)** by implementing a parallel counting program on a **distributed-memory** model. Each process independently generates its own data, counts occurrences of a target value locally, and then combines results across all processes using `MPI_Reduce`.  
The lab also compares MPI with shared-memory threading (Pthreads) and analyzes speedup/efficiency.

## Files
- `count_mpi.c` – Parallel counting program: each rank generates random data, counts target occurrences locally, and uses `MPI_Reduce` to compute the global total and timing.

## Challenges Faced
- **Non‑linear speedup / overhead effects:**  
  Speedup was not perfectly linear as processes increased because MPI has overhead (process startup + reduce communication). Also, randomness and system load can affect timing results.  
  **Resolution:** Use larger workloads, run multiple trials, and average timings for more stable performance measurements.

- **Result variation between runs:**  
  The total count differs slightly across runs because each process generates random arrays independently.  
  **Resolution:** Use a fixed seed per process (e.g., `srand(rank)`) if repeatable results are required.

- **Efficiency calculation interpretation:**  
  Efficiency can drop as process count increases due to communication and synchronization costs.  
  **Resolution:** Interpreted efficiency as a practical indicator of scaling limits rather than expecting perfect scaling.

#include <stdio.h>
#include <mpi.h>

int main(int argc, char *argv[]) {

    int mynode, totalnodes;
    int sum, startval, endval, accum;
    MPI_Status status;
    double start_time, end_time;

    MPI_Init(&argc, &argv);
    MPI_Comm_size(MPI_COMM_WORLD, &totalnodes);
    MPI_Comm_rank(MPI_COMM_WORLD, &mynode);

    MPI_Barrier(MPI_COMM_WORLD);
    start_time = MPI_Wtime();

    sum = 0;

    startval = 1000 * mynode / totalnodes + 1;
    endval   = 1000 * (mynode + 1) / totalnodes;

    for (int i = startval; i <= endval; i++) {
        sum += i;
    }

    if (mynode != 0) {
        MPI_Send(&sum, 1, MPI_INT, 0, 1, MPI_COMM_WORLD);
    } else {
        for (int j = 1; j < totalnodes; j++) {
            MPI_Recv(&accum, 1, MPI_INT, j, 1, MPI_COMM_WORLD, &status);
            sum += accum;
        }
    }

    end_time = MPI_Wtime();

    if (mynode == 0) {
        printf("Total sum = %d\n", sum);
        printf("Execution time with %d processes: %f seconds\n",
               totalnodes, end_time - start_time);
    }

    MPI_Finalize();
    return 0;
}

scatter_gather.c


#include <stdio.h>
#include <stdlib.h>
#include <mpi.h>

#define N 1000

int main(int argc, char** argv) {

    int rank, size;
    int *array = NULL;
    int *local_array;
    int local_n;
    int local_sum = 0, total_sum = 0;

    MPI_Init(&argc, &argv);

    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    local_n = N / size;

    local_array = (int*) malloc(local_n * sizeof(int));

    if (rank == 0) {
        array = (int*) malloc(N * sizeof(int));

        for (int i = 0; i < N; i++)
            array[i] = i + 1;
    }

    MPI_Scatter(array, local_n, MPI_INT, local_array, local_n, MPI_INT, 0, MPI_COMM_WORLD);

    for (int i = 0; i < local_n; i++)
        local_sum += local_array[i];

    printf("Process %d: local sum = %d\n", rank, local_sum);

    MPI_Reduce(&local_sum, &total_sum, 1, MPI_INT, MPI_SUM, 0, MPI_COMM_WORLD);

    int *all_sums = NULL;

    if (rank == 0)
        all_sums = (int*) malloc(size * sizeof(int));

    MPI_Gather(&local_sum, 1, MPI_INT, all_sums, 1, MPI_INT, 0, MPI_COMM_WORLD);

    if (rank == 0) {

        printf("Total sum via MPI_Reduce: %d\n", total_sum);

        printf("Collected sums: ");
        for (int i = 0; i < size; i++)
            printf("%d ", all_sums[i]);

        printf("\nExpected total sum = %d\n", N * (N + 1) / 2);

        free(array);
        free(all_sums);
    }

    free(local_array);

    MPI_Finalize();
    return 0;
}


scatterv_gatherv.c

#include <stdio.h>
#include <stdlib.h>
#include <mpi.h>

#define N 1000

int main(int argc, char** argv) {

    int rank, size;
    int *array = NULL;
    int *local_array;

    int *sendcounts = NULL;
    int *displs = NULL;

    int local_n;
    int local_sum = 0;
    int total_sum = 0;

    int *all_sums = NULL;

    MPI_Init(&argc, &argv);

    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    if (rank == 0) {

        array = (int*) malloc(N * sizeof(int));
        sendcounts = (int*) malloc(size * sizeof(int));
        displs = (int*) malloc(size * sizeof(int));

        for (int i = 0; i < N; i++)
            array[i] = i + 1;

        int sum = 0;

        for (int i = 0; i < size; i++) {

            sendcounts[i] = N / size;

            if (i < N % size)
                sendcounts[i]++;

            displs[i] = sum;
            sum += sendcounts[i];
        }
    }

    MPI_Scatter(sendcounts, 1, MPI_INT, &local_n, 1, MPI_INT, 0, MPI_COMM_WORLD);

    local_array = (int*) malloc(local_n * sizeof(int));

    MPI_Scatterv(array, sendcounts, displs, MPI_INT,
                 local_array, local_n, MPI_INT,
                 0, MPI_COMM_WORLD);

    for (int i = 0; i < local_n; i++)
        local_sum += local_array[i];

    printf("Process %d: local sum = %d\n", rank, local_sum);

    MPI_Reduce(&local_sum, &total_sum, 1, MPI_INT, MPI_SUM, 0, MPI_COMM_WORLD);

    if (rank == 0)
        all_sums = (int*) malloc(size * sizeof(int));

    MPI_Gatherv(&local_sum, 1, MPI_INT,
                all_sums, sendcounts, displs,
                MPI_INT, 0, MPI_COMM_WORLD);

    if (rank == 0) {

        printf("Total sum = %d\n", total_sum);
        printf("Expected sum = %d\n", N * (N + 1) / 2);

        printf("Collected local sums:\n");

        for (int i = 0; i < size; i++)
            printf("%d ", all_sums[i]);

        printf("\n");

        free(array);
        free(sendcounts);
        free(displs);
        free(all_sums);
    }

    free(local_array);

    MPI_Finalize();
    return 0;
}


matvec.c

#include <stdio.h>
#include <stdlib.h>
#include <mpi.h>

#define M 8
#define N 8

int main(int argc, char** argv) {

    int rank, size;

    double *A = NULL;
    double *local_A;

    double *x = NULL;
    double *local_y;
    double *y = NULL;

    int local_rows;

    MPI_Init(&argc, &argv);

    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    if (M % size != 0) {

        if (rank == 0)
            printf("Rows must be divisible by processes\n");

        MPI_Finalize();
        return 1;
    }

    local_rows = M / size;

    local_A = (double*) malloc(local_rows * N * sizeof(double));
    local_y = (double*) malloc(local_rows * sizeof(double));

    if (rank == 0) {

        A = (double*) malloc(M * N * sizeof(double));
        x = (double*) malloc(N * sizeof(double));
        y = (double*) malloc(M * sizeof(double));

        for (int i = 0; i < M * N; i++)
            A[i] = 1.0;

        for (int i = 0; i < N; i++)
            x[i] = 1.0;

    } else {

        x = (double*) malloc(N * sizeof(double));
    }

    MPI_Bcast(x, N, MPI_DOUBLE, 0, MPI_COMM_WORLD);

    MPI_Scatter(A, local_rows * N, MPI_DOUBLE,
                local_A, local_rows * N, MPI_DOUBLE,
                0, MPI_COMM_WORLD);

    for (int i = 0; i < local_rows; i++) {

        local_y[i] = 0.0;

        for (int j = 0; j < N; j++)
            local_y[i] += local_A[i * N + j] * x[j];
    }

    MPI_Gather(local_y, local_rows, MPI_DOUBLE,
               y, local_rows, MPI_DOUBLE,
               0, MPI_COMM_WORLD);

    if (rank == 0) {

        printf("Result vector y:\n");

        for (int i = 0; i < M; i++)
            printf("%f ", y[i]);

        printf("\n");

        free(A);
        free(y);
    }

    free(local_A);
    free(local_y);
    free(x);

    MPI_Finalize();

    return 0;
}

Broadcast.c

#include <mpi.h>
#include <stdio.h>

int main(int argc, char** argv) {
    int rank, size;
    int data = 100;
    double start, end;

    MPI_Init(&argc, &argv);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    MPI_Barrier(MPI_COMM_WORLD);
    start = MPI_Wtime();

    // MPI_Bcast
    MPI_Bcast(&data, 1, MPI_INT, 0, MPI_COMM_WORLD);

    MPI_Barrier(MPI_COMM_WORLD);
    end = MPI_Wtime();

    if (rank == 0)
        printf("MPI_Bcast Time: %f\n", end - start);

    // Manual broadcast
    MPI_Barrier(MPI_COMM_WORLD);
    start = MPI_Wtime();

    if (rank == 0) {
        for (int i = 1; i < size; i++) {
            MPI_Send(&data, 1, MPI_INT, i, 0, MPI_COMM_WORLD);
        }
    } else {
        MPI_Recv(&data, 1, MPI_INT, 0, 0, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
    }

    MPI_Barrier(MPI_COMM_WORLD);
    end = MPI_Wtime();

    if (rank == 0)
        printf("Manual Broadcast Time: %f\n", end - start);

    MPI_Finalize();
    return 0;
}

TreeReduce.c

#include <mpi.h>
#include <stdio.h>

int main(int argc, char** argv) {
    int rank, size;
    int value, step;

    MPI_Init(&argc, &argv);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    value = rank + 1;

    for (step = 1; step < size; step *= 2) {
        if (rank % (2 * step) == 0) {
            if (rank + step < size) {
                int temp;
                MPI_Recv(&temp, 1, MPI_INT, rank + step, 0, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
                value += temp;
            }
        } else {
            MPI_Send(&value, 1, MPI_INT, rank - step, 0, MPI_COMM_WORLD);
            break;
        }
    }

    if (rank == 0)
        printf("Final Reduced Sum = %d\n", value);

    MPI_Finalize();
    return 0;
}

    MPI_Finalize();

    MPI_Comm_size(MPI_COMM_WORLD, &size);

    sum = value;

    if (rank != 0) {
        MPI_Recv(&sum, 1, MPI_INT, rank - 1, 0, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
        sum += value;
    }

    if (rank != size - 1) {

}

    int rank, size;
    MPI_Comm_rank(comm, &rank);
    MPI_Comm_size(comm, &size);

    if (rank == root) {
        for (int i = 0; i < size; i++) {
            if (i == root) {
                recvbuf[i] = sendbuf[0];
            } else {
                MPI_Recv(&recvbuf[i], 1, MPI_INT, i, 0, comm, MPI_STATUS_IGNORE);

#include <mpi.h>
#include <stdio.h>

int main(int argc, char** argv) {
    int rank, size;
    int local[4], prefix[4];
    int sum = 0, offset;

    MPI_Init(&argc, &argv);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    // Example data
    for (int i = 0; i < 4; i++)
        local[i] = rank * 4 + i + 1;

    // Local prefix sum
    prefix[0] = local[0];
    for (int i = 1; i < 4; i++)
        prefix[i] = prefix[i - 1] + local[i];

    sum = prefix[3];

    // Get offset using scan
    MPI_Scan(&sum, &offset, 1, MPI_INT, MPI_SUM, MPI_COMM_WORLD);

    offset -= sum;

    for (int i = 0; i < 4; i++)
        prefix[i] += offset;

    printf("Rank %d: ", rank);
    for (int i = 0; i < 4; i++)
        printf("%d ", prefix[i]);
    printf("\n");

    MPI_Finalize();
    return 0;
}
